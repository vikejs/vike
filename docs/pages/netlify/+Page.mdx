### 2. Install

Install the Netlify Vite adapter:

```shell
npm install @netlify/vite-plugin
```

```js
// vite.config.js

import vike from 'vike/plugin'
import netlify from '@netlify/vite-plugin' // [!code ++]

export default {
  plugins: [
    vike(),
    netlify({ // [!code ++]
      build: { // [!code ++]
        enabled: true // [!code ++]
      } // [!code ++]
    }) // [!code ++]
 ]
}
```

> The build command remains `$ vike build`.
